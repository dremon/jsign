042    /**

044     *

054        /** The position of the PE header in the file */
055        private final long peHeaderOffset;
056    
057        private final File file;
058        private final ExtendedRandomAccessFile raf;
059    
060        public PEFile(File file) throws IOException {
061            this.file = file;
062            raf = new ExtendedRandomAccessFile(file, "rw");
063            
064            // DOS Header
065            
066            byte[] buffer = new byte[2];
067            raf.read(buffer);
068            
069            if (!Arrays.equals(buffer, "MZ".getBytes())) {
070                throw new IOException("DOS header signature not found");
071            }
072            
073            raf.seek(0x3C);
074            peHeaderOffset = raf.readDWord();
075            
076            // PE Header
077            
078            raf.seek(peHeaderOffset);
079            
080            buffer = new byte[4];
081            raf.read(buffer);
082            if (!Arrays.equals(buffer, new byte[] { 'P', 'E', 0, 0})) {
083                throw new IOException("PE signature not found as expected at offset 0x" + Long.toHexString(peHeaderOffset));
084            }
085        }
086

088            raf.close();
089        }

378        /**
379         * Compute the checksum of the image file. The algorithm for computing
380         * the checksum is incorporated into IMAGHELP.DLL.
381         */
382        public synchronized long computeChecksum() {
383            PEImageChecksum checksum = new PEImageChecksum(peHeaderOffset + 88);
384            
385            byte[] b = new byte[64 * 1024];
386            
387            try {
388                raf.seek(0);
389                
390                int len;
391                while ((len = raf.read(b)) > 0) {
392                    checksum.update(b, 0, len);
393                }
394            } catch (IOException e) {
395                throw new RuntimeException(e);
396            }
397            
398            return checksum.getValue();
399        }

401        public synchronized void updateChecksum() {
402            ByteBuffer buffer = ByteBuffer.allocate(4);
403            buffer.order(ByteOrder.LITTLE_ENDIAN);
404            buffer.putInt((int) computeChecksum());
405            
406            try {            
407                raf.seek(peHeaderOffset + 88);
408                raf.write(buffer.array());
409            } catch (IOException e) {
410                throw new RuntimeException(e);
411            }
412        }

502        /**
503         * Writes the data directory of the specified type. The data is appended
504         * at the end of the file. If a previous entry was already present it's
505         * left as is in the file and is simply dereferenced from the table. 
506         * 
507         * @param type
508         * @param data
509         * @throws IOException
510         */
511        public synchronized void writeDataDirectory(DataDirectoryType type, byte[] data) throws IOException {
512            // todo overwrite an existing entry at the end of the file
513            
514            // append the data directory at the end of the file
515            long offset = raf.length();        
516            
517            raf.seek(offset);
518            raf.write(data);
519            
520            // add the entry in the data directory table
521            DataDirectory entry = getDataDirectory(type);
522            entry.write(offset, data.length);
523            
524            updateChecksum();
525        }

527        /**
528         * Returns the authenticode signatures on the file.
529         */
530        public synchronized List<CMSSignedData> getSignatures() {
531            List<CMSSignedData> signatures = new ArrayList<CMSSignedData>();
532            
533            for (CertificateTableEntry entry : getCertificateTable()) {
534                try {
535                    signatures.add(entry.getSignature());
536                } catch (UnsupportedOperationException e) {
537                    // unsupported type, just skip
538                } catch (Exception e) {
539                    e.printStackTrace();
540                }
541            }
542            
543            return signatures;
544        }

546        private synchronized List<CertificateTableEntry> getCertificateTable() {
547            List<CertificateTableEntry> entries = new ArrayList<CertificateTableEntry>();
548            DataDirectory certificateTable = getDataDirectory(DataDirectoryType.CERTIFICATE_TABLE);
549            
550            if (certificateTable != null && certificateTable.exists()) {
551                long position = certificateTable.getVirtualAddress();
552                
553                try {
554                    entries.add(new CertificateTableEntry(this, position));
555                    
556                    // todo read the remaining signatures
557                } catch (Exception e) {
558                    e.printStackTrace();
559                }
560            }
561            
562            return entries;
563        }

583        /**
584         * Print detailed informations about the PE file.
585         */
586        public void printInfo(PrintWriter out) {
587            out.println("PE File");
588            out.println("  Name:          " + file.getName());
589            out.println("  Size:          " + file.length());
590            out.println("  Last Modified: " + new Date(file.lastModified()));
591            out.println();
592            
593            out.println("PE Header");
594            out.println("  Machine:                    " + getMachineType());
595            out.println("  Number of sections:         " + getNumberOfSections());
596            out.println("  Timestamp:                  " + getTimeDateStamp());
597            out.println("  Pointer to symbol table:    0x" + Long.toHexString(getPointerToSymbolTable()));
598            out.println("  Number of symbols:          " + getNumberOfSymbols());
599            out.println("  Size of optional header:    " + getSizeOfOptionalHeader());
600            out.println("  Characteristics:            0x" + Long.toBinaryString(getCharacteristics()));
601            out.println();
602            
603            out.println("Optional Header");
604            PEFormat format = getFormat();
605            out.println("  PE Format:                  0x" + Integer.toHexString(format.value) + " (" + format.label + ")");
606            out.println("  Linker version:             " + getMajorLinkerVersion() + "." + getMinorLinkerVersion());
607            out.println("  Size of code:               " + getSizeOfCode());
608            out.println("  Size of initialized data:   " + getSizeOfInitializedData());
609            out.println("  Size of uninitialized data: " + getSizeOfUninitializedData());
610            out.println("  Address of entry point:     0x" + Long.toHexString(getAddressOfEntryPoint()));
611            out.println("  Base of code:               0x" + Long.toHexString(getBaseOfCode()));
612            if (PEFormat.PE32.equals(getFormat())) {
613                out.println("  Base of data:               0x" + Long.toHexString(getBaseOfData()));
614            }
615            out.println("  Image base:                 0x" + Long.toHexString(getImageBase()));
616            out.println("  Section alignment:          " + getSectionAlignment());
617            out.println("  File alignment:             " + getFileAlignment());
618            out.println("  Operating system version:   " + getMajorOperatingSystemVersion() + "." + getMinorOperatingSystemVersion());
619            out.println("  Image version:              " + getMajorImageVersion() + "." + getMinorImageVersion());
620            out.println("  Subsystem version:          " + getMajorSubsystemVersion() + "." + getMinorSubsystemVersion());
621            out.println("  Size of image:              " + getSizeOfImage());
622            out.println("  Size of headers:            " + getSizeOfHeaders());
623            out.println("  Checksum:                   0x" + Long.toHexString(getCheckSum()));
624            out.println("  Checksum (computed):        0x" + Long.toHexString(computeChecksum()));
625            out.println("  Subsystem:                  " + getSubsystem());
626            out.println("  DLL characteristics:        0x" + Long.toBinaryString(getDllCharacteristics()));
627            out.println("  Size of stack reserve:      " + getSizeOfStackReserve());
628            out.println("  Size of stack commit:       " + getSizeOfStackCommit());
629            out.println("  Size of heap reserve:       " + getSizeOfHeapReserve());
630            out.println("  Size of heap commit:        " + getSizeOfHeapCommit());
631            out.println("  Number of RVA and sizes:    " + getNumberOfRvaAndSizes());
632            out.println();
633            
634            out.println("Data Directory");
635            for (DataDirectoryType type : DataDirectoryType.values()) {
636                DataDirectory entry = getDataDirectory(type);
637                if (entry != null && entry.exists()) {
638                    out.printf("  %-30s 0x%08x %8d bytes\n", type, entry.getVirtualAddress(), entry.getSize());
639                }
640            }
641            out.println();
642            
643            out.println("Sections");
644            out.println("      Name     Virtual Size  Virtual Address  Raw Data Size  Raw Data Ptr  Characteristics");
645            List<Section> sections = getSections();
646            for (int i = 0; i < sections.size(); i++) {
647                Section section = sections.get(i);
648                out.printf("  #%d  %-8s     %8d       0x%08x       %8d    0x%08x  %s\n", i + 1, section.getName(), section.getVirtualSize(), section.getVirtualAddress(), section.getSizeOfRawData(), section.getPointerToRawData(), section.getCharacteristics());
649            }
650            out.println();
651            
652            List<CMSSignedData> signatures = getSignatures();
653            if (!signatures.isEmpty()) {
654                out.println("Signatures");
655                for (CMSSignedData signedData : signatures) {
656                    SignerInformation signerInformation = signedData.getSignerInfos().getSigners().iterator().next();
657                    X509CertificateHolder certificate = (X509CertificateHolder) signedData.getCertificates().getMatches(signerInformation.getSID()).iterator().next();
658                    
659                    String commonName = certificate.getSubject().getRDNs(X509ObjectIdentifiers.commonName)[0].getFirst().getValue().toString();
660                    
661                    AttributeTable unsignedAttributes = signerInformation.getUnsignedAttributes();
662                    boolean timestamped = unsignedAttributes != null &&
663                               (unsignedAttributes.get(PKCSObjectIdentifiers.pkcs_9_at_counterSignature) != null
664                             || unsignedAttributes.get(AuthenticodeObjectIdentifiers.SPC_RFC3161_OBJID)  != null);
665                    DigestAlgorithm algorithm = DigestAlgorithm.of(signerInformation.getDigestAlgorithmID().getAlgorithm());
666                    out.println("  " + commonName + "  " + (algorithm != null ? "[" + algorithm.id + "]  " : "") + (timestamped ? "(timestamped)" : ""));
667                }
668            }
669        }

671        /**
672         * Compute the digest of the file. The checksum field, the certificate
673         * directory table entry and the certificate table are excluded from
674         * the digest.
675         */
676        private synchronized byte[] computeDigest(MessageDigest digest) throws IOException {
677            long checksumLocation = peHeaderOffset + 88;
678            
679            DataDirectory certificateTable = getDataDirectory(DataDirectoryType.CERTIFICATE_TABLE);
680            
681            // digest from the beginning to the checksum field (excluded)
682            updateDigest(digest, 0, checksumLocation);
683            
684            // skip the checksum field
685            long position = checksumLocation + 4;
686            
687            // digest from the end of the checksum field to the beginning of the certificate table entry
688            int certificateTableOffset = getDataDirectoryOffset() + 8 * DataDirectoryType.CERTIFICATE_TABLE.ordinal();
689            updateDigest(digest, position, certificateTableOffset);
690            
691            // skip the certificate entry
692            position = certificateTableOffset + 8;
693            
694            // todo digest the sections in ascending address order
695            
696            // digest from the end of the certificate table entry to the beginning of the certificate table
697            if (certificateTable != null && certificateTable.exists()) {
698                updateDigest(digest, position, certificateTable.getVirtualAddress());
699                position = certificateTable.getVirtualAddress() + certificateTable.getSize();
700            }
701            
702            // digest from the end of the certificate table to the end of the file
703            updateDigest(digest, position, raf.length());
704            
705            return digest.digest();
706        }

708        /**
709         * Update the specified digest by reading the underlying RandomAccessFile
710         * from the start offset included to the end offset excluded.
711         * 
712         * @param digest
713         * @param startOffset
714         * @param endOffset
715         */
716        private void updateDigest(MessageDigest digest, long startOffset, long endOffset) throws IOException {
717            raf.seek(startOffset);
718            
719            byte[] buffer = new byte[8192];
720            
721            long position = startOffset;
722            while (position < endOffset) {
723                int length = (int) Math.min(buffer.length, endOffset - position);
724                raf.read(buffer, 0, length);
725                
726                digest.update(buffer, 0, length);
727                
728                position += length;
729            }
730        }
731    
732        /**
733         * Compute the checksum of the file using the specified digest algorithm.
734         * 
735         * @param algorithm the digest algorithm, typically SHA1
736         */
737        public byte[] computeDigest(DigestAlgorithm algorithm) throws IOException {
738            return computeDigest(algorithm.getMessageDigest());
739        }

741        /**
742         * Increase the size of the file up to a size that is a multiple of the specified value.
743         * 
744         * @param multiple
745         */
746        public synchronized void pad(int multiple) throws IOException {
747            long padding = (multiple - raf.length() % multiple) % multiple;
748            raf.seek(raf.length());
749            for (int i = 0; i < padding; i++) {
750                raf.writeByte(0);
751            }
752        }
